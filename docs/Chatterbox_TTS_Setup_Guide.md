```python
class ChatterboxEngine(TTSEngine):
    """Chatterbox TTS engine using pyttsx3 - high-quality cross-platform voice synthesis"""
    
    def __init__(self):
        super().__init__("chatterbox")  # üëà Named "chatterbox"
        self.engine = None
        
        # Define available voice configurations
        self.voice_configs = {
            "default": {"rate": 200, "volume": 0.9, "voice_id": 0},
            "female_1": {"rate": 180, "volume": 0.9, "voice_id": 0},
            "male_1": {"rate": 200, "volume": 0.9, "voice_id": 1},
            "expressive": {"rate": 220, "volume": 1.0, "voice_id": 0},
            "calm": {"rate": 160, "volume": 0.8, "voice_id": 0}
        }
    
    def _check_availability(self):
        try:
            import pyttsx3  # üëà Uses pyttsx3 engine
            self.engine = pyttsx3.init(debug=False)
            voices = self.engine.getProperty('voices')
            if voices:
                self.available = True
                logger.info("‚úÖ Chatterbox TTS (pyttsx3) engine available with voice support")
        except ImportError as e:
            logger.warning(f"‚ùå Chatterbox TTS not available: {e}")
            self.available = False
```

### **Voice Synthesis Process**

```python
async def synthesize(self, text: str, voice_id: str = "default", speed: float = 1.0) -> bytes:
    # Get voice configuration
    config = self.voice_configs.get(voice_id, self.voice_configs["default"])
    
    # Configure voice properties
    voices = self.engine.getProperty('voices')
    if voices and len(voices) > config["voice_id"]:
        self.engine.setProperty('voice', voices[config["voice_id"]].id)
    
    # Apply speed and volume
    rate = int(config["rate"] * speed)
    self.engine.setProperty('rate', rate)
    self.engine.setProperty('volume', config["volume"])
    
    # Generate audio to temporary file
    with tempfile.NamedTemporaryFile(suffix='.wav', delete=False) as tmp_file:
        logger.info(f"üéµ Chatterbox synthesizing - voice: {voice_id}, rate: {rate}")
        
        # Save to file and wait for completion
        self.engine.save_to_file(text, tmp_file.name)
        self.engine.runAndWait()
        
        # Read generated audio
        with open(tmp_file.name, 'rb') as f:
            audio_data = f.read()
        
        # Clean up temporary file
        os.unlink(tmp_file.name)
        
        logger.info(f"‚úÖ Chatterbox synthesis successful - {len(audio_data)} bytes")
        return audio_data
```

```python
def get_service_urls():
    """Get service URLs based on whether we're running in Docker or locally."""
    if not is_docker:
        # Local development
        tts_url = "http://localhost:8005"  # üëà Chatterbox TTS service
    else:
        # Docker environment
        tts_url = os.getenv("TTS_SERVICE_URL", "http://tts-service:8005")
    
    return whisper_url, tts_url

@router.post("/synthesize")
async def synthesize_speech(
    request: TTSRequest,
    current_user: User = Depends(get_current_user)
):
    """Convert text to speech using Chatterbox TTS service."""
    _, tts_url = get_service_urls()
    
    async with httpx.AsyncClient() as client:
        response = await client.post(
            f"{tts_url}/synthesize",  # üëà Calls Chatterbox TTS service
            json={
                "text": request.text,
                "voice_id": request.voice_id,
                "speed": request.speed,
                "format": request.format
            }
        )
        
        if response.status_code == 200:
            return StreamingResponse(
                io.BytesIO(response.content),
                media_type="audio/wav",
                headers={"X-TTS-Engine": "chatterbox"}
            )
```

## 8. üéØ **Why Chatterbox is Primary**

### **Engine Priority Order**

1. **ü•á Chatterbox** - High-quality, cross-platform, offline-capable
2. **ü•à Edge TTS** - Microsoft neural voices (requires internet)
3. **ü•â gTTS** - Google voices (requires internet)
4. **üèÖ eSpeak** - Basic offline fallback
5. **üîá Silence** - Ultimate fallback (generates silence)

### **Key Advantages**

- ‚úÖ **Offline Operation**: Works without internet connection
- ‚úÖ **Voice Variety**: 5 distinct voice personalities
- ‚úÖ **Cross-Platform**: Works on Windows, macOS, Linux
- ‚úÖ **Speed Control**: Dynamic rate adjustment (0.5x - 2.0x)
- ‚úÖ **Volume Control**: Per-voice volume optimization
- ‚úÖ **Reliable Fallback**: Never fails to produce audio
